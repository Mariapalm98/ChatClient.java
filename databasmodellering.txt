Create table user(
id int primary key auto_increment,
name varcher(70) not null,
password varcher(100) not null,
prfilbild varcher(200) not null (url),
beskrivning text
);

Create table message(
id int primary key auto_increment,
userId int not null,
text text not null,
timestamp created_at defult current_timestamp,
foreign key (userId) refrences User(id)
);

Create table kommentar(
id int primary key auto_increment,
messageId int not null,
userId int not null,
text text not null,
foreign key (messageId) references message(id),
foreign key (userId) references user(id)
);
